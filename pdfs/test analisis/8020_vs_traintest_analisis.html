data_of_8020 = test_vs_8020[test_vs_8020['split80/20']=='yes']
data_of_train_n_test = test_vs_8020[test_vs_8020['split80/20']=='no']

# Accuracy = TP+TN/TP+FP+FN+TN
Accuracy_8020 = (data_of_8020['TP'] + data_of_8020['TN']) /\
        (data_of_8020['TP'] + data_of_8020['TN'] + data_of_8020['FP'] + data_of_8020['FN'])

Accuracy_test = (data_of_train_n_test['TP'] + data_of_train_n_test['TN'] )/\
        (data_of_train_n_test['TP'] + data_of_train_n_test['TN'] + data_of_train_n_test['FP'] + data_of_train_n_test['FN'])

# Precision = TP/TP+FP
Precision_8020 = data_of_8020['TP'] / (data_of_8020['TP'] + data_of_8020['FP'])
Precision_test = data_of_train_n_test['TP'] /( data_of_train_n_test['TP'] + data_of_train_n_test['FP'])

# Recall = TP/TP+FN
Recall_8020 = data_of_8020['TP'] / (data_of_8020['TP'] + data_of_8020['FN'])
Recall_test = data_of_train_n_test['TP'] / (data_of_train_n_test['TP'] + data_of_train_n_test['FN'])

# F1_Score = 2*(Recall * Precision) / (Recall + Precision)
F1_score_8020 = 2 * (Recall_8020 * Precision_8020) / (Recall_8020 + Precision_8020)
F1_score_test = 2 * (Recall_test * Precision_test) / (Recall_test + Precision_test)

test_vs_8020.loc[test_vs_8020['split80/20'] == 'yes', 'Accuracy'] = Accuracy_8020
test_vs_8020.loc[test_vs_8020['split80/20'] == 'no', 'Accuracy'] = Accuracy_test
test_vs_8020.loc[test_vs_8020['split80/20'] == 'yes', 'Precision'] = Precision_8020
test_vs_8020.loc[test_vs_8020['split80/20'] == 'no', 'Precision'] = Precision_test
test_vs_8020.loc[test_vs_8020['split80/20'] == 'yes', 'Recall'] = Recall_8020
test_vs_8020.loc[test_vs_8020['split80/20'] == 'no', 'Recall'] = Recall_test
test_vs_8020.loc[test_vs_8020['split80/20'] == 'yes', 'F1_Score'] = F1_score_8020
test_vs_8020.loc[test_vs_8020['split80/20'] == 'no', 'F1_Score'] = F1_score_test
